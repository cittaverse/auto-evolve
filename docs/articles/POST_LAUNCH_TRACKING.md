# 用法：bash scripts/track-article-impact.sh [文章发布日期]

ARTICLE_DATE="${1:-2026-03-17}"

# ... 其他仓库

# 输出 JSON
echo "{\"date\": \"$(date -I)\", \"article\": \"auto-evolve\", \"stars\": ...}" > $OUTPUT_FILE

- T+1d 起每日自动执行 (cron)
