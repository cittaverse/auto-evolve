#### 修复 1: 情绪词汇多样性检测

```python
# 1. 添加情绪词汇多样性检测
def check_emotion_diversity(emotion_words: List[str]) -> float:
    """
    检测情绪词汇多样性
    返回：多样性系数 (0-1)
    """
    if len(emotion_words) == 0:
        return 1.0
    
    unique_ratio = len(set(emotion_words)) / len(emotion_words)
    
    # 重复度惩罚：如果同一个词出现超过 3 次，降低多样性系数
    from collections import Counter
    counter = Counter(emotion_words)
    max_repeat = max(counter.values())
    
    if max_repeat > 3:
        repeat_penalty = 1.0 / (max_repeat / 3)
    else:
        repeat_penalty = 1.0
    
    return unique_ratio * repeat_penalty


# 2. 在情绪唤醒度计算中应用多样性惩罚
def _compute_arousal_score(self, ...):
    # ... 原有计算 ...
    
    # 新增：多样性惩罚
    emotion_words_list = self._extract_emotion_words(text)
    diversity = self.check_emotion_diversity(emotion_words_list)
    
    # 应用惩罚 (多样性越低，惩罚越大)
    score = score * (0.5 + 0.5 * diversity)
    
    return max(1.0, min(5.0, score))
```
